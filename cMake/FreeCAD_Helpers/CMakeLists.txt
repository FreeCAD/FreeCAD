file(GLOB FREECAD_HELPER_CMAKE_FILES ./*.cmake)
set(_SKIP_FILES "SetupQt.cmake")
foreach(_CMAKE_FILE ${FREECAD_HELPER_CMAKE_FILES})

    endif()
endforeach(_CMAKE_FILE ${FREECAD_HELPER_CMAKE_FILES})
