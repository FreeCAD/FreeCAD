    add_definitions(-DHAVE_LIMITS_H -DHAVE_CONFIG_H)

    ${CMAKE_BINARY_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${Boost_INCLUDE_DIRS}
    ${OCC_INCLUDE_DIR}
    ${PYTHON_INCLUDE_DIRS}
    ${ZLIB_INCLUDE_DIR}
    ${XercesC_INCLUDE_DIRS}
)

    FreeCADApp

    Part

    InspectionFeature.cpp

    PreCompiled.cpp
    PreCompiled.h

    ../Init.py
)
